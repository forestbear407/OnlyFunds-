// Handle login
const loginForm = document.getElementById('loginForm');
if (loginForm) {
    loginForm.addEventListener('submit', function(event) {
        event.preventDefault();
        const username = document.getElementById('username').value;
        document.getElementById('loginMessage').innerText = `Welcome, ${username}! You are now logged in.`;
    });
}

// Handle wishlist
const wishlistForm = document.getElementById('wishlistForm');
if (wishlistForm) {
    wishlistForm.addEventListener('submit', function(event) {
        event.preventDefault();
        const name = document.getElementById('itemName').value;
        const link = document.getElementById('itemLink').value;

        const listItem = document.createElement('li');
        listItem.innerHTML = `<a href="${link}" target="_blank">${name}</a>`;
        document.getElementById('wishlistItems').appendChild(listItem);

        wishlistForm.reset();
    });
}
